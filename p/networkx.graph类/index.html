# 节点属性操作，先要保证 1 结点存在
G.nodes[1]['color'] = 'blue'
G.nodes.add(2, 'color' = 'blue')

color = G.nodes[1].get('color', 'default')
color = G.nodes.get(1, 'color', 'default')

# 边属性操作
G.edges[1, 2]['weight'] = 5.0
weight = G.edges[1, 2].get('weight', 1.0)

注意点：

  1. 语法等效性：属性名 color（关键字参数）和 'color'（字典键）本质是同一个属性。

  2. 安全操作：用 G.nodes.get(2, {}) 可避免异常。

import networkx as nx

# 创建图
G = nx.Graph(name="示例图")
G.add_nodes_from([1, 2, 3], category='sample')
G.add_edges_from([(1, 2), (2, 3)], weight=1.0)

# 设置属性
G.nodes[1]['color'] = 'red'
G.edges[1, 2]['weight'] = 2.5

# 输出信息
print("节点列表:", list(G.nodes))
print("边列表:", list(G.edges(data=True)))
print("节点1的邻居:", list(G.neighbors(1)))
print("节点1的度数:", G.degree(1))

注意事项

  • 边操作会自动添加不存在的节点
  • 节点支持任意可哈希对象（如元组、字符串）
  • 批量操作比逐条操作更高效
