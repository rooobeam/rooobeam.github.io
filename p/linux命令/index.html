# -*- coding: utf-8 -*-
import pdb
def add(a, b):
    return a + b

if __name__ == '__main__':
    print("===start===")
    pdb.set_trace()
    c = add(1, 3)
    print("===end===")

    (Pdb) l （“小写L指令”：查看）（若连续输入两次，则第二次返回[EOF]）

    (Pdb) p 变量名 （查看当前可查变量）
