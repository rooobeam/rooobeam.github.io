class Solution:
    def maxSum(self, nums: List[int], m: int, k: int) -> int:
        sum_n = max_sum = 0
        dif_n=defaultdict(int)

        for i, element in enumerate(nums):
            dif_n[element] += 1
            sum_n += element
            
            left = i-k+1			
            if left < 0:
                continue

            if len(dif_n) >= m:
                max_sum = max(sum_n,max_sum)

            out=nums[left]
            sum_n -= out
            dif_n[out] -= 1
            if dif_n[out] == 0:
                del dif_n[out]
			
        return max_sum
